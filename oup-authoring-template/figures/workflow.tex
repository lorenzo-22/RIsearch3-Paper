% RIsearch3 system workflow — TikZ source

  comp/.style={

  },
  param/.style={
    draw=ParamBorder, fill=ParamGray,

  },
  marrow/.style={->, thick, draw=black!65},

% ── INPUT ────────────────────────────────────────────────────────────────────

\node[inbox] (input) {
  siRNA FASTA\\[2pt]
  Target FASTA\\[2pt]
  GTF\,/\,BED
};

% ── STAGE 1: RIsearch (index + search) ───────────────────────────────────────

\node[comp, right=0.55cm of input] (search) {
  \textbf{Stage\,1}\\[2pt]
  RIsearch
};

\node[tool, below=0.65cm of search] (t-search) {
  \texttt{risearch.index()}\\[2pt]
  \texttt{risearch.search()}
};

\node[param, below=0.3cm of t-search] (p-search) {
  PyO3 bindings\\Polars DataFrame
};

% ── STAGE 2: Intersection ────────────────────────────────────────────────────

\node[comp, right=0.55cm of search] (isect) {
  \textbf{Stage\,2}\\[2pt]
  Intersection
};

\node[tool, below=0.65cm of isect] (t-isect) {
  \texttt{IntersectionService}\\[1pt]
  \texttt{TranscriptomeParser}
};

\node[param, below=0.3cm of t-isect] (p-isect) {
  NCLS interval index\\gw\,/\,tw mode
};

% ── GENOME FASTA INPUT (feeds Stage 3 via L-arrow) ───────────────────────────

\node[genomebox, above=0.7cm of isect] (genome) {
  Genome\\FASTA
};

% ── STAGE 3: Accessibility (on-the-fly ViennaRNA) ────────────────────────────

\node[comp, right=0.55cm of isect] (acc) {
  \textbf{Stage\,3}\\[2pt]
  Accessibility
};

\node[tool, below=0.65cm of acc] (t-acc) {
  \texttt{GenomeAccessibility}\\[-1pt]\texttt{Service}\\[1pt]
  ViennaRNA\,2.7.2
};

\node[param, below=0.3cm of t-acc] (p-acc) {
  $W{=}80,\,L{=}40,\,u{=}30$\\[2pt]$O_i{=}{-}RT\ln p_i$
};

% ── STAGE 4: Partition function ──────────────────────────────────────────────

\node[comp, right=0.55cm of acc] (prob) {
  \textbf{Stage\,4}\\[2pt]
  Partition\\Function
};

\node[tool, below=0.65cm of prob] (t-prob) {
  \texttt{ProbabilityService}\\[1pt]
  Polars \texttt{group\_by}
};

\node[param, below=0.3cm of t-prob] (p-prob) {
  $W_i{=}A_i\,e^{-(E^\prime_i{+}O_i)/RT}$\\[2pt]$\alpha,\gamma,\theta$ sweeps
};

% ── OUTPUT ───────────────────────────────────────────────────────────────────

\node[outbox, right=0.55cm of prob] (output) {
  $p_{\mathrm{off},i}$\\per site\\[3pt]
  $P_{\mathrm{OFF}}$\\aggregate\\[3pt]
  TSV\,$\cdot$\,CSV\\Parquet
};

% ── SHARED TEMPERATURE PARAMETER (above Stage 4) ─────────────────────────────

\node[tempbox, above=0.7cm of prob] (temp) {

};
